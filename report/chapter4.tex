\chapter{Requirement Specification}
\label{chap:requirements}

\section{Functional Requirements}

\begin{enumerate}[label=FR-\arabic*.]
  \item \textbf{Sensor Monitoring:} The system shall continuously sample tri-axial
        acceleration from the MPU6050 at a minimum rate of 50~Hz and compute a
        1-second RMS value.
  \item \textbf{Wave Classification:} The system shall classify each 1-second window
        into one of three states: \textsc{calm} ($a_{\rm rms} < 0.03$~g),
        \textsc{moderate} ($0.03 \leq a_{\rm rms} < 0.075$~g), or
        \textsc{high-wave} ($a_{\rm rms} \geq 0.075$~g).
  \item \textbf{GPS Tagging:} Each sensor reading shall be tagged with GPS
        coordinates read from the NEO-6M module (latitude, longitude, UTC time).
  \item \textbf{WiFi Dashboard:} The system shall expose an HTTP REST API on the
        local WiFi network and update a React-based dashboard in real time using
        periodic polling or WebSocket push.
  \item \textbf{GSM SMS Alert:} Upon entering \textsc{high-wave} state, the system
        shall dispatch an SMS alert to all pre-configured recipient numbers within
        12~seconds via the SIM800L module.
  \item \textbf{Data Persistence:} The FastAPI backend shall persist every sensor
        record (timestamp, $a_x$, $a_y$, $a_z$, rms, state, lat, lon) in SQLite.
  \item \textbf{Admin Control:} An authenticated admin panel shall allow enabling/
        disabling SMS alerts and viewing registered recipient numbers.
  \item \textbf{System Status:} The dashboard shall display live system status
        including WiFi signal strength, GPS fix status, and battery voltage.
\end{enumerate}

\section{Non-Functional Requirements}

\begin{enumerate}[label=NF-\arabic*.]
  \item \textbf{Response Time:} Sensor-to-classification latency shall be
        $<$100~ms. WiFi dashboard update latency shall be $<$500~ms. SMS
        delivery shall complete within 12~seconds.
  \item \textbf{Accuracy:} Acceleration readings shall have an error $\leq\pm$0.01~g
        post-calibration. GPS position error shall be $\leq$5~m under clear sky.
  \item \textbf{Availability:} The system shall maintain $>$99\% uptime during the
        6--8~hour operational window without manual intervention.
  \item \textbf{Battery Endurance:} The system shall operate for $\geq$6~hours
        continuously on a single fully charged 2\,200~mAh 18650 cell.
  \item \textbf{Power Safety:} The system shall incorporate over-charge, over-discharge,
        and short-circuit protection for the Li-ion cell.
  \item \textbf{Usability:} The dashboard UI shall be accessible on any modern browser
        without requiring a client-side installation.
  \item \textbf{Portability:} The firmware shall target the Arduino/ESP32 toolchain
        and compile without modification on any ESP32 DevKit~v1 board.
  \item \textbf{Security:} The admin panel shall require password authentication
        (JWT-based); sensor data endpoints are read-only and unauthenticated.
\end{enumerate}

\section{Software Requirements}

\begin{enumerate}[label=SW-\arabic*.]
  \item \textbf{Arduino IDE 1.8.19} with ESP32 board package v3.3.7 installed via
        \texttt{boards.json} URL:
        \texttt{https://raw.githubusercontent.com/espressif/arduino-esp32/gh-pages/package\_esp32\_index.json}.
  \item \textbf{Arduino Libraries:}
        \begin{itemize}
          \item \texttt{Adafruit\_MPU6050} v2.2.4 (sensor driver)
          \item \texttt{TinyGPS++} v1.0.3 (NMEA parser)
          \item \texttt{WiFi.h} (bundled with ESP32 core)
          \item \texttt{HTTPClient.h} (bundled with ESP32 core)
          \item \texttt{SoftwareSerial.h} (bundled with ESP32 core)
        \end{itemize}
  \item \textbf{Python 3.11+} with:
        \begin{itemize}
          \item \texttt{fastapi==0.115.x}
          \item \texttt{uvicorn==0.30.x}
          \item \texttt{sqlalchemy==2.0.x}
          \item \texttt{pydantic==2.x}
          \item \texttt{python-jose} (JWT authentication)
        \end{itemize}
  \item \textbf{Node.js 20+} with \texttt{react==18.x}, \texttt{vite==5.x},
        \texttt{axios} for the frontend.
  \item \textbf{SQLite 3.x} (embedded; no separate server required).
\end{enumerate}

\section{Communication Requirements}

\begin{itemize}
  \item \textbf{WiFi:} IEEE 802.11~b/g/n at 2.4~GHz; effective operational range
        $\approx$50~m line-of-sight between buoy and shore station.
  \item \textbf{GSM:} SIM800L on 900/1800~MHz bands; SIM card with SMS credit
        required; SMS delivery 8--12~s from trigger.
  \item \textbf{GPS:} NEO-6M UART at 9\,600~baud; open-sky first fix $\leq$30~s.
  \item \textbf{I2C:} 400~kHz fast mode; GPIO21 (SDA), GPIO22 (SCL); 4.7~k$\Omega$
        pull-ups.
  \item \textbf{UART (GPS):} GPIO16 (RX2 $\leftarrow$ NEO-6M TX), GPIO17 (TX2 –
        not used).
  \item \textbf{UART (SIM800L):} GPIO27 (TX2$\rightarrow$SIM800L RX),
        GPIO26 (RX2$\leftarrow$SIM800L TX); baud 9\,600.
\end{itemize}
